This post is about Decision Tree Regression, currently just its implementation in R. Hopefully in the coming days I will include the theory of the method. The data was acquired from superdatascience. They do a great job of teaching the topic as well.

# Install and load packages
required_packages <- c("tidyverse", "data.table", 
                       "caTools", "e1071", "rpart",
                       "rmarkdown", "knitr")

packageCheck <- lapply(required_packages, FUN = function(x) {
  if(!require(x, character.only = TRUE)) {
    install.packages(x, dependencies = TRUE)
    library(x, character.only = TRUE)
  }
})
salary_decision_tree <- rpart(Salary ~ Level, data = salary_data, control = rpart.control(minsplit = 1))
x_grid = data.frame(Level = seq(min(salary_data$Level), max(salary_data$Level), 0.01))
pred_x_grid = predict(salary_decision_tree, x_grid)
ggplot() +
  geom_point(aes(x = salary_data$Level, y = salary_data$Salary), colour = "black") +
  geom_line(aes(x = x_grid$Level, y = pred_x_grid), colour = "red") +
  xlab("Level") +
  ylab("Salary") +
  theme_classic()
