This post is about Random Forests Regression, currently just its implementation in R. Hopefully in the coming days I will include the theory of the method. The data was acquired from superdatascience. They do a great job of teaching the topic as well.

# Install and load packages
required_packages <- c("tidyverse", "data.table", 
                       "caTools", "e1071", "rpart", "randomForest",
                       "rmarkdown", "knitr")

packageCheck <- lapply(required_packages, FUN = function(x) {
  if(!require(x, character.only = TRUE)) {
    install.packages(x, dependencies = TRUE)
    library(x, character.only = TRUE)
  }
})
# Create the random forests model
salary_rf <- randomForest::randomForest(Salary ~ Level, data = salary_data, ntree = 10)
# Create a grid for predictions
x_grid <- data.frame(Level = seq(min(salary_data$Level), max(salary_data$Level), 0.01))
y_pred <- predict(salary_rf, x_grid)
# Data visualization
ggplot() +
  geom_point(aes(x = salary_data$Level, salary_data$Salary)) +
  geom_line(aes(x = x_grid$Level, y = y_pred)) + 
  xlab("Level") +
  ylab("Salary") +
  theme_classic()
