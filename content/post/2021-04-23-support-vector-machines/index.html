# Fit a SVM model to the data
svm_salary <- svm(Salary ~ Level, data = salary_data, scale = T, type = "eps-regression")
x_grid = data.frame(Level = seq(min(salary_data$Level), max(salary_data$Level), 0.5))
pred_x_grid = predict(svm_salary, x_grid)
ggplot() +
  geom_point(aes(salary_data$Level, salary_data$Salary), colour = "black") +
  geom_line(aes(x_grid$Level, pred_x_grid), colour = "red") +
  xlab("Position") +
  ylab("Salary") +
  theme_classic()
